## C++에서 배열을 특정한 수로 초기화하고 싶을 때

~~
fill_n({배열}, {배열 크기}, {특정한 수});
~~~

## c++ 배열 초기화

~~~
//100은 size

//0으로 초기화

int arr[100] = {};
int arr[100] = {0};
int arr[100] = {0,};
int arr[100][100] = {};
int arr[100][100] = {0};
int arr[100][100] = {0,};

//특정값으로 초기화

//fill_n, 1차원 배열
fill_n(arr, 100, 2); //모든 값을 2로 초기화

//memset, sring.h, 2차원 배열에서 0이나 1로 초기화하는 경우만 가능
memset(arr, -1, sizeof(arr);

//fill
fill(&arr[0][0], &arr[100][100], 2); //모든 값을 2로 초기화
~~~
